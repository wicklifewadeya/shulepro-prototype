npx @better-auth/cli generate --config ./lib/auth.ts

# generate the Prisma Client after schema changes
npx prisma generate


# open Prisma Studio (GUI to browse / edit your data)
npx prisma studio


# create + apply a migration in development (recommended dev flow)
npx prisma migrate dev --name descriptive_name


# apply pending migrations in production / CI
npx prisma migrate deploy


# push the Prisma schema directly to the DB (no migration files)
npx prisma db push


# pull an existing DB schema into schema.prisma (introspection)
npx prisma db pull


# reset the development DB (drops data and reapplies migrations)
npx prisma migrate reset


# mark migrations as applied / resolve migration history manually
npx prisma migrate resolve --applied MIGRATION_NAME


# run your seed script (configured in package.json under "prisma".seed)
npx prisma db seed
